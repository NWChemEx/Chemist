152 template<typename NucleiType>
153 typename NucleiUnion<NucleiType>::reference NucleiUnion<NucleiType>::get_nuke_(
154  size_type i) {
155  // Trying to return from the loop yields a compiler warning, about missing
156  // return type, so we move the return outside the loop.
157  size_type j = 0;
158  for(; j < m_nuclei_.size(); ++j) {
159  if(m_nuclei_[j].size() > i) break;
160  i -= m_nuclei_[j].size();
161  }
162  return m_nuclei_[j][i];
163 }
