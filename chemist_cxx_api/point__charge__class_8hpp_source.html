184template<typename LHSType, typename RHSType>
185bool operator==(const PointCharge<LHSType>& lhs,
186 const PointCharge<RHSType>& rhs) {
187 if constexpr(!std::is_same_v<LHSType, RHSType>) {
188 return false;
189 } else {
190 const Point<LHSType>& lhs_point = lhs;
191 const Point<RHSType>& rhs_point = rhs;
192 return (lhs.charge() == rhs.charge()) && (lhs_point == rhs_point);
193 }
194}
