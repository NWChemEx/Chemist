173inline void ConnectivityTablePIMPL::set_natoms(size_type natoms) {
174 if(natoms == m_natoms_) return;
175 std::vector<bool> temp(natoms * natoms, false);
176 m_connections_.swap(temp);
177 m_natoms_ = natoms;
178}

180inline auto ConnectivityTablePIMPL::nbonds() const noexcept {
181 size_type n_bonds = 0;
182 for(size_type i = 0; i < m_natoms_; ++i)
183 for(size_type j = i + 1; j < m_natoms_; ++j)
184 if(m_connections_[i * m_natoms_ + j]) ++n_bonds;
185 return n_bonds;
186}

188inline bool ConnectivityTablePIMPL::are_bonded(size_type i, size_type j) const {
189 auto [min, max] = sanitize_indices_(i, j);
190 return m_connections_[min * m_natoms_ + max];
191}

193inline void ConnectivityTablePIMPL::add_bond(size_type i, size_type j) {
194 auto [min, max] = sanitize_indices_(i, j);
195 m_connections_[min * m_natoms_ + max] = true;
196}

198inline typename ConnectivityTablePIMPL::offset_pair_list
199ConnectivityTablePIMPL::bonds() const {
200 const auto n_bonds = nbonds();
201 offset_pair_list rv(n_bonds);
202
203 for(size_type i = 0, bonds_found = 0; i < m_natoms_; ++i) {
204 for(size_type j = i + 1; j < m_natoms_; ++j) {
205 if(are_bonded(i, j)) {
206 rv[bonds_found++] = offset_pair{i, j};
207 if(bonds_found == n_bonds) return rv;
208 }
209 }
210 }
211 // We can get here if there's actually zero bonds...
212 return rv;
213}

215inline void ConnectivityTablePIMPL::bounds_check_(size_type atom_i) const {
216 if(atom_i < m_natoms_) return;
217
218 const auto msg = std::to_string(atom_i) + " is greater than the number " +
219 "atoms (" + std::to_string(m_natoms_) + ").";
220 throw std::out_of_range(msg);
221}

223inline typename ConnectivityTablePIMPL::offset_pair
224ConnectivityTablePIMPL::sanitize_indices_(size_type i, size_type j) const {
225 auto min = std::min(i, j);
226 auto max = std::max(i, j);
227 bounds_check_(min);
228 bounds_check_(max);
229 if(min == max) {
230 std::string msg = "Atom indices can not be equal! (i == j == ";
231 msg += std::to_string(i) + ").";
232 throw std::out_of_range(msg);
233 }
234 return offset_pair{min, max};
235}
