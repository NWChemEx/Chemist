56 void add_atomic_basis_set(const_reference c) {
57 m_shells_per_center_.push_back(c.size());
58 m_shell_offsets_.push_back(n_shells());
59
60 m_names_.push_back(c.basis_set_name());
61 m_atomic_numbers_.push_back(c.atomic_number());
62
63 m_x_.push_back(c.center().x());
64 m_y_.push_back(c.center().y());
65 m_z_.push_back(c.center().z());

67 for(const auto& shell_i : c) add_shell(shell_i);

82 auto shell_range(size_type center) const {
83 size_type begin = m_shell_offsets_[center];
84 size_type end = begin + m_shells_per_center_[center];
85 return typename abs_traits::range_type{begin, end};
86 }

88 auto primitive_range(size_type shell) const {
89 size_type begin = m_primitive_offset_[shell];
90 size_type end = begin + m_primitives_per_shell_[shell];
91 return typename abs_traits::range_type{begin, end};
92 }

94 size_type shell_to_center(size_type shell) const {
95 for(size_type i = 0; i < size(); ++i) {
96 auto [shell_begin, shell_end] = shell_range(i);
97 if(shell >= shell_begin && shell < shell_end) return i;
98 }
99 throw std::runtime_error("Non-existent shell requested.");
100 }
101
102 size_type primitive_to_center(size_type primitive) const {
103 return shell_to_center(primitive_to_shell(primitive));
104 }
105
106 size_type primitive_to_shell(size_type primitive) const {
107 for(size_type i = 0; i < n_shells(); ++i) {
108 auto [p_begin, p_end] = primitive_range(i);
109 if(primitive >= p_begin && primitive < p_end) return i;
110 }
111 throw std::runtime_error("Non-existent primitive requested.");
112 }

126 auto at(size_type i) {
127 using reference_type = typename abs_traits::shell_reference;
128 std::vector<reference_type> shells;
129 shells.reserve(m_shells_per_center_[i]);
130
131 auto [shell_begin, shell_end] = shell_range(i);
132 for(auto shell_i = shell_begin; shell_i < shell_end; ++shell_i) {
133 shells.emplace_back(std::move(shell(shell_i)));
134 }
135
136 return reference(m_names_[i], m_atomic_numbers_[i], center(i),
137 std::move(shells));
138 }

222private:
224 void add_shell(typename abs_traits::const_shell_reference s) {
225 m_pure_.push_back(s.pure());
226 m_l_.push_back(s.l());
227
228 const auto prims_in_shell = s.n_primitives();
229
230 m_primitives_per_shell_.push_back(prims_in_shell);
231 m_primitive_offset_.push_back(n_primitives());
232 for(size_type i = 0; i < prims_in_shell; ++i) {
233 add_primitive(s.primitive(i));
234 }
235 }
236
238 void add_primitive(typename abs_traits::const_primitive_reference p) {
239 m_coefs_.push_back(p.coefficient());
240 m_exps_.push_back(p.exponent());
241 }
