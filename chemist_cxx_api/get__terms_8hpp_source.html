63template<typename ReturnType, typename VisitorType, typename OperatorType>
64ReturnType get_terms(OperatorType&& container) {
65 using clean_type = std::decay_t<OperatorType>;
66 using value_type = typename clean_type::value_type;
67 using container_type = std::vector<value_type>;
68 container_type ops;
69 auto size = container.size();
70 for(decltype(size) i = 0; i < size; ++i) {
71 const auto& op = container.get_operator(i);
72 VisitorType visitor;
73 op.visit(visitor);
74 if(visitor.m_is_good_term) {
75 ops.emplace_back(container.coefficient(i), op.clone());
76 }
77 }
78 return ReturnType(std::move(ops));
79}
