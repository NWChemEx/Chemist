56 ContractedGaussianPIMPL(coefficient_vector cs, exponent_vector exps,
57 typename cg_traits::center_type r0) :
58 m_center_(std::move(r0)),
59 m_coefs_(std::move(cs)),
60 m_exp_(std::move(exps)) {}

62 // -------------------------------------------------------------------------

65
67 void add_primitive(typename cg_traits::coefficient_type c,
68 typename cg_traits::exponent_type e) {
69 m_coefs_.reserve(m_coefs_.size() + 1);
70 m_exp_.reserve(m_exp_.size() + 1);
71 m_coefs_.push_back(c);
72 m_exp_.push_back(e);
73 }

103 return m_exp_[i];
104 }
105
107 auto operator[](size_type i) noexcept {

114 using prim_ref = typename cg_traits::const_primitive_reference;
115 return prim_ref(coefficient(i), exponent(i), center());

117
119 auto size() const noexcept { return m_coefs_.size(); }
120
121 // -------------------------------------------------------------------------
122 // -- Utility
123 // -------------------------------------------------------------------------
124
126 bool operator==(const ContractedGaussianPIMPL& rhs) const noexcept {
127 auto lhs_state = std::tie(m_center_, m_coefs_, m_exp_);

130 }
