530template<typename PrimitiveType>
531template<typename PrimType2, typename>
532PrimitiveView<PrimitiveType>& PrimitiveView<PrimitiveType>::operator=(
533 const PrimType2& rhs) {
534 assert_non_null_();
535 if(rhs.is_null())
536 throw std::runtime_error("Assigning a null Primitive to a non-null "
537 "PrimitiveView is currently not "
538 "supported.");
539 center().x() = rhs.center().x();
540 center().y() = rhs.center().y();
541 center().z() = rhs.center().z();
542 coefficient() = rhs.coefficient();
543 exponent() = rhs.exponent();
544 return *this;
545}

547// Ensures PrimitiveType can appear on the left side of operator==
548template<typename PrimitiveType>
549bool operator==(const std::decay_t<PrimitiveType>& lhs,
550 const PrimitiveView<PrimitiveType>& rhs) {
551 return rhs == lhs;
552}

554// Ensures PrimitiveType can appear on the left side of operator!=
555template<typename PrimitiveType>
556bool operator!=(const std::decay_t<PrimitiveType>& lhs,
557 const PrimitiveView<PrimitiveType>& rhs) {
