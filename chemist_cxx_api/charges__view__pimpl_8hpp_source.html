165template<typename ChargesType>
166template<typename DerivedType>
167bool ChargesViewPIMPL<ChargesType>::are_equal_impl_(
168 const ChargesViewPIMPL& other) const noexcept {
169 const auto* pthis = dynamic_cast<const DerivedType*>(this);
170 assert(pthis != nullptr); // Coding error if this happens
171 const auto* pother = dynamic_cast<const DerivedType*>(&other);
172
173 // Same derived type can just delegate to operator==
174 if(pother != nullptr) { return (*pthis) == (*pother); }
175
176 // Need to manually check through common API.
177
178 // Short-circuit if different sizes
179 if(size() != other.size()) return false;
180
181 // Compare elements in the range [0, size()), n.b. is no-op if size() == 0
182 for(size_type i = 0; i < size(); ++i)
183 if((*this)[i] != other[i]) return false;
184
185 // Getting here means they're the same
186 return true;
187}
