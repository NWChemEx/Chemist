473 template<typename NucleiType>
474 template<typename BeginItr, typename EndItr>
475 void FragmentedNuclei<NucleiType>::insert(BeginItr begin, EndItr end) {
476  // If we put the arguments directly into a vector this is what would result
477  using vec_t = decltype(std::vector(begin, end));
478 
479  // These would be the elements
480  using input_type = typename vec_t::value_type;
481 
482  // Were we given indices or nuclei?
483  if constexpr(std::is_integral_v<input_type>) {
484  nucleus_index_set inputs(begin, end);
485  insert(std::move(inputs));
486  } else {
487  // Assumed to be Nucleus or NucleusView objects
488  using input_t = typename const_reference::reference_container;
489  input_t inputs(begin, end);
490  insert(const_reference(inputs));
491  }
492 }
