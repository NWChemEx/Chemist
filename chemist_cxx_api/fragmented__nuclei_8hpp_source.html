477template<typename NucleiType>
478template<typename BeginItr, typename EndItr>
479void FragmentedNuclei<NucleiType>::insert(BeginItr begin, EndItr end) {
480 // If we put the arguments directly into a vector this is what would result
481 using vec_t = decltype(std::vector(begin, end));
482
483 // These would be the elements
484 using input_type = typename vec_t::value_type;
485
486 // Were we given indices or nuclei?
487 if constexpr(std::is_integral_v<input_type>) {
488 nucleus_index_set inputs(begin, end);
489 insert(std::move(inputs));
490 } else {
491 // Assumed to be Nucleus or NucleusView objects
492 using input_t = typename const_reference::reference_container;
493 input_t inputs(begin, end);
494 insert(const_reference(inputs));
495 }
496}
