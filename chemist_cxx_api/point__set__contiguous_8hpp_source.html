136template<typename PointSetType>
137bool PointSetContiguous<PointSetType>::operator==(
138 const PointSetContiguous& rhs) const noexcept {
139 // Must have the same number of points to be equal
140 if(m_n_points_ != rhs.m_n_points_) return false;
141
142 // N.b. From here on we know both have the same number of points
143
144 // If there's zero points the addresses may be junk and we don't want to
145 // compare them, so short circuit if empty
146 if(m_n_points_ == 0) return true;
147
148 // Try to short circuit if the addresses are the same
149 const auto plhs = std::tie(m_px_, m_py_, m_pz_);
150 const auto prhs = std::tie(rhs.m_px_, rhs.m_py_, rhs.m_pz_);
151 if(plhs == prhs) return true;
152
153 // Now we have to actually compare the values
154 return std::equal(m_px_, m_px_ + m_n_points_, rhs.m_px_) &&
155 std::equal(m_py_, m_py_ + m_n_points_, rhs.m_py_) &&
156 std::equal(m_pz_, m_pz_ + m_n_points_, rhs.m_pz_);
157}
