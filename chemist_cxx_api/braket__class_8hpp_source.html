388template<typename BraType, typename OperatorType, typename KetType>
389template<typename BraType2, typename OperatorType2, typename KetType2>
390bool BraKet<BraType, OperatorType, KetType>::operator==(
391 const BraKet<BraType2, OperatorType2, KetType2>& rhs) const noexcept {
392 constexpr bool same_bra_type = std::is_same_v<BraType, BraType2>;
393 constexpr bool same_op_type = std::is_same_v<OperatorType, OperatorType2>;
394 constexpr bool same_ket_type = std::is_same_v<KetType, KetType2>;
395
396 if constexpr(same_bra_type && same_op_type && same_ket_type) {
397 return bra() == rhs.bra() && op().are_equal(rhs.op()) &&
398 ket() == rhs.ket();
399 } else {
400 return false;
401 }
