656template<typename AtomicBasisSetType>
657template<typename ABSType2, typename>
658AtomicBasisSetView<AtomicBasisSetType>&
659AtomicBasisSetView<AtomicBasisSetType>::operator=(const ABSType2& rhs) {
660 assert_non_null_();
661 if(rhs.is_null())
662 throw std::runtime_error(
663 "Assigning a null AtomicBasisSet to a non-null AtomicBasisSet is "
664 "currently not supported");
665 if(rhs.size() != this->size())
666 throw std::runtime_error(
667 "AtomicBasisSet to assign must be the same size as the aliased one");
668
669 basis_set_name() = rhs.basis_set_name();
670 atomic_number() = rhs.atomic_number();
671 center().x() = rhs.center().x();
672 center().y() = rhs.center().y();
673 center().z() = rhs.center().z();
674 for(size_type shell = 0; shell < this->size(); ++shell) {
675 (*this)[shell] = rhs[shell];
676 }
677 return *this;
678}
