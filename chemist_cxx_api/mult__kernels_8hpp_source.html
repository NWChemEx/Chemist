68  return std::make_pair(true, result);

107  static return_type eval(ResultType&& result, LHSType&& lhs, RHSType&& rhs) {
108  auto lhs_variant = lhs.variant(result);
109  auto rhs_variant = rhs.variant(result);
110  auto result_variant = result.variant(result);

112  auto l = [&](auto&& result) {
