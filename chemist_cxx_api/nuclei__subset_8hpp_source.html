150 template<typename NucleiType>
151 inline bool NucleiSubset<NucleiType>::operator==(
152  const NucleiSubset& rhs) const noexcept {
153  if(this->size() != rhs.size()) return false;
154  if(this->size() == 0) return true;
155 
156  // If supersystems are the same just compare the members arrays
157  if(m_nuclei_ == rhs.m_nuclei_) { return m_members_ == rhs.m_members_; }
158 
159  // different supersystems means we need to compare the values one by one
160  for(size_type i = 0; i < this->size(); ++i)
161  if(this->get_nuke(i) != rhs.get_nuke(i)) return false;
162 
163  return true;
164 }
