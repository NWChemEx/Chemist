88 auto grab_diagonal(TensorType&& t) {
89  auto trange1 = t.trange().dim(0);
90  if(trange1 != t.trange().dim(1))
91  throw std::runtime_error("Expected a square tiling");
92  TA::TiledRange trange{{trange1}};
93  using tensor_type = std::decay_t<TensorType>;
94  using tile_type = typename tensor_type::value_type;
95  t.world().gop.fence();
96  auto l = [=](tile_type& tile, const TA::Range& r) {
97  auto idx = get_block_idx(trange, r);
98  auto t_tile = t.find({idx[0], idx[0]}).get();
99  tile = tile_type(r);
100  for(auto i : r) {
101  std::vector ii{i[0], i[0]};
102  tile[i] = t_tile[ii];
103  }
104  return tile.norm();
105  };
106  auto rv = TA::make_array<tensor_type>(t.world(), trange, l);
107  t.world().gop.fence();
108  return rv;

233 template<typename TileType, typename PolicyType, typename AddOp,
234  typename TimesOp, typename ResultType>
235 auto reduce_tot_elementwise(const TA::DistArray<TileType, PolicyType>& lhs,
236  const TA::DistArray<TileType, PolicyType>& rhs,
237  AddOp&& add_op, TimesOp&& times_op,
238  ResultType&& init) {
239  using inner_type = typename TileType::value_type;
240  using scalar_type = typename TileType::scalar_type;
241  auto inner_times = [times_op{std::forward<TimesOp>(times_op)}](
242  ResultType& result, const scalar_type& first,

303 template<typename T, typename U,
304  typename V = typename std::decay_t<T>::scalar_type>
305 bool allclose(T&& actual, U&& ref, const bool abs_comp = false,
306  V&& rtol = 1.0E-5, V&& atol = 1.0E-8,
307  std::size_t inner_rank = 0) {
308  using tensor_type = std::decay_t<T>;
309  using tile_type = typename tensor_type::value_type;
310  using scalar_type = std::decay_t<V>;
311  static_assert(std::is_same_v<tensor_type, std::decay_t<U>>,
312  "different tensor types is currently unsupported");
313 
314  constexpr bool is_tot = TA::detail::is_tensor_of_tensor_v<tile_type>;
315 
316  // Get a dummy string label (something like "i0, i1, i2, ...")
317  auto idx = TA::detail::dummy_annotation(actual.range().rank(), inner_rank);
318 
319  // Compute A - B, call result AmB
320  tensor_type AmB;
321  if(abs_comp) {
322  // TODO: There probably is some way to do this without forming the
323  // intermediate tensors
324  auto abs_lambda = [](const scalar_type& val) { return std::fabs(val); };
325  auto abs_actual = apply_elementwise(actual, abs_lambda);
326  auto abs_ref = apply_elementwise(ref, abs_lambda);
327  AmB(idx) = abs_actual(idx) - abs_ref(idx);
328  } else {
329  AmB(idx) = actual(idx) - ref(idx);
330  }
331 
332  auto times_op = [=](scalar_type lhs, scalar_type rhs) {
333  return std::fabs(lhs) <= atol + rtol * std::fabs(rhs);
334  };
335 
336  std::logical_and<bool> add_op;
337 
338  if constexpr(!is_tot) {
339  return reduce_elementwise(AmB, ref, add_op, times_op, true).get();
340  } else {
341  using inner_type = typename tile_type::value_type;
342  auto inner_times = [=](bool& result, const scalar_type& first,
343  const scalar_type& second) {
344  result = times_op(first, second);
345  };
346  auto outer_times = [=](inner_type lhs, inner_type rhs) {
347  return lhs.reduce(rhs, inner_times, add_op, true);
348  };
349  return reduce_elementwise(AmB, ref, add_op, outer_times, true,
350  inner_rank)
351  .get();
352  }
353 }

405 template<typename LHSTileType, typename LHSPolicyType, typename RHSTileType,

407 bool operator==(const TA::DistArray<LHSTileType, LHSPolicyType>& lhs,
408  const TA::DistArray<RHSTileType, RHSPolicyType>& rhs) {
409  if constexpr(!std::is_same_v<decltype(lhs), decltype(rhs)>)
410  return false;
411  else {
412  if(lhs.is_initialized() != rhs.is_initialized()) return false;
413  if(!lhs.is_initialized()) return true;
414  if(lhs.trange() != rhs.trange()) return false;
415 
416  using scalar_type = typename LHSTileType::scalar_type;
417  using namespace chemist::ta_helpers;
418  std::logical_and<bool> add_op;
419  std::equal_to<scalar_type> times_op;
420  if constexpr(!TA::detail::is_tensor_of_tensor_v<LHSTileType>) {
421  return reduce_elementwise(lhs, rhs, add_op, times_op, true);
422  } else {
423  return reduce_tot_elementwise(lhs, rhs, add_op, times_op, true);
424  }
