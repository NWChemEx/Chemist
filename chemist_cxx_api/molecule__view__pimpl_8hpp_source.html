93template<typename MoleculeType>
94bool MoleculeViewPIMPL<MoleculeType>::operator==(
95 const MoleculeViewPIMPL& rhs) const noexcept {
96 if(m_nuclei_ != rhs.m_nuclei_) return false;
97 if(m_nuclei_.size() == 0) return true;
98
99 // Factor out logic to compare the pointers
100 auto compare_pointers = [](auto&& plhs, auto&& prhs) {
101 if(plhs != prhs) {
102 // In theory shouldn't be nullptrs, but we check anyways
103 if(plhs == nullptr || prhs == nullptr) return false;
104 if(*plhs != *prhs) return false;
105 return true;
106 }
107 return true;
108 };
109 if(!compare_pointers(m_pnet_electrons_, rhs.m_pnet_electrons_))
110 return false;
111 return compare_pointers(m_pmultiplicity_, rhs.m_pmultiplicity_);
112}
