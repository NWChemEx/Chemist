29 inline auto uninject_index_(const ElementIndex& idx,
30  const std::map<std::size_t, std::size_t>& injections) {
31  const auto r = idx.size();
32  if(injections.empty()) return idx;
33  for(const auto& [k, v] : injections)
34  if(k >= r)
35  throw std::runtime_error("Mode: " + std::to_string(k) +
36  " is not in range [0, " +
37  std::to_string(r) + ")");
38  std::vector<std::size_t> uninjected_idx(r - injections.size());
39  for(std::size_t i = 0, counter = 0; i < r; ++i){
40  if(!injections.count(i)) {
41  uninjected_idx[counter] = idx[i];
42  ++counter;
43  }
44  }
45  return ElementIndex(std::move(uninjected_idx));
46 }

89  inner_tile_t buffer(TA::Range(d.result_extents()));
90

93  for(const auto& [k, v] : ind2mode) injections[v] = oeidx[k];

98  inner_tile_t t = tensor.find(itidx);
99 
100  // It's not clear to me whether the injection alters the order. If
101  // the indices in injected_d are ordered such that the i-th index
102  // of injected_d is the i-th index of d (with the former containing
103  // the injected modes) then we can loop over injected_d, d zipped
104  // together and avoid the uninjection

109  // map it to output
