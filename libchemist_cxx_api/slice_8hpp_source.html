30 inline bool is_empty_slice_(const sparse_map::ElementIndex& lo,
31  const sparse_map::ElementIndex& hi) {
32  const auto rank = lo.size();
33 
34  // lo and hi must have same rank
35  TA_ASSERT(rank == hi.size());
36  // Make sure hi is greater than equal to lo, count modes that are equal
37  std::size_t nequal = 0;
38  for(const auto& [l, h] : utilities::Zip(lo, hi)) {
39  TA_ASSERT(h >= l);
40  if(h == l) ++nequal;
41  }
42 
43  // The slice is empty if all modes are equal
44  const bool is_empty = nequal == rank;
45 
46  // Either all of or none of the modes are allowed to be equal. Anything in
47  // between reduces the rank, which we're not supporting at the moment.
48  TA_ASSERT(nequal == 0 || is_empty);
49 
50  return is_empty;
51 }

103 template<typename TensorType>
104 auto slice(const TensorType& t, const sparse_map::ElementIndex& lo,
105  const sparse_map::ElementIndex& hi) {
106  const auto& tr = t.trange();
107  const auto rank = tr.rank();
108  const auto annotation = TA::detail::dummy_annotation(rank);
109 
110  // We rely on error checking done in is_empty_slice_ for everything except
111  // making sure lo and hi are the same rank as the tensor, and that lo and
112  // hi are in the tensor
113  TA_ASSERT(rank == lo.size());
114  const bool is_empty = detail_::is_empty_slice_(lo, hi);
115  TA_ASSERT(!is_empty); // TODO: This function should handle empty slices
116  TA_ASSERT(tr.elements_range().includes(lo));
117 
118  // hi is just past the last element we want, hi "minus 1" is the last
119  // element we want.
120  sparse_map::ElementIndex hi_m1(hi);
121  for(auto& x : hi_m1) x -= 1;
122  TA_ASSERT(tr.elements_range().includes(hi_m1));
123 
124  // Now we need to determine if lo/hi are the leading/ending elements of
125  // their respective tiles
126  const bool lo_good = is_tile_lower_bound(tr, lo);
127  const bool hi_good = is_tile_upper_bound(tr, hi);
128 
129  // If we don't have to retile this is easy, just return the slice
130  if(lo_good && hi_good) {
131  auto [tlo, thi] = detail_::get_slice_tile_indices_(tr, lo, hi_m1);
132  TensorType rv;
133  rv(annotation) = t(annotation).block(tlo, thi);
134  return rv;
135  }
136 
137  // Getting here means we need to retile before grabbing the block
138  auto new_tr = insert_tile_boundaries(tr, lo, hi);
139  const TensorType new_t = TA::retile(t, new_tr);
140  auto [tlo, thi] = detail_::get_slice_tile_indices_(new_tr, lo, hi_m1);
141  TensorType rv;
142  rv(annotation) = new_t(annotation).block(tlo, thi);
143  return rv;
144 }
