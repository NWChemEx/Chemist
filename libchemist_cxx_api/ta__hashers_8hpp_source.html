100 template<typename TensorType, typename PolicyType>
101 pluginplay::HashValue get_tile_hash_sum(
102  const TA::DistArray<TensorType, PolicyType>& A) {
103  auto& madworld = A.world();
104  auto& mpiworld = madworld.mpi.Get_mpi_comm();
105 
106  // Note: Without the fence orbital space hash tests hang on parallel runs.
107  madworld.gop.fence();
108  pluginplay::HashValue myhash;
109  pluginplay::HashValue mytotal(16, 0); // pluginplay::HashType::Hash128 has
110  // 16 uint8_t
111  pluginplay::HashValue hashsum(16, 0);
112 
113  for(auto it = A.begin(); it != A.end(); ++it) {
114  auto tile = A.find(it.index()).get();
115  myhash = pluginplay::make_hash(pluginplay::HashType::Hash128, tile);
116  for(auto i = 0; i < myhash.size(); i++) { mytotal[i] += myhash[i]; }
117  }
118  if(madworld.size() > 1 && !A.pmap().get()->is_replicated()) {
119  // Note: Using &mytotal, &hashsum gives Seg Fault 11.
120  madworld.mpi.Allreduce(mytotal.data(), hashsum.data(), mytotal.size(),
121  MPI_UINT8_T, MPI_SUM);
122  } else {
123  hashsum = mytotal;
124  }
125  madworld.gop.fence();
126  return hashsum;
127 }
